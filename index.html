# ========================================== # SCANNER COIN BARU MENYENTUH MA200 DAILY # Binance Spot # ========================================== import requests import pandas as pd import ta from datetime import datetime BASE_URL = "https://api.binance.com" # Ambil semua pair USDT exchange_info = requests.get(f"{BASE_URL}/api/v3/exchangeInfo").json() symbols = [ s["symbol"] for s in exchange_info["symbols"] if s["quoteAsset"] == "USDT" and s["status"] == "TRADING" ] results = [] for symbol in symbols: try: url = f"{BASE_URL}/api/v3/klines" params = { "symbol": symbol, "interval": "1d", "limit": 250 } data = requests.get(url, params=params, timeout=10).json() if len(data) < 200: continue df = pd.DataFrame(data) df.columns = [ "open_time","open","high","low","close", "volume","close_time","qav", "trades","tbav","tqav","ignore" ] df["close"] = df["close"].astype(float) df["high"] = df["high"].astype(float) df["low"] = df["low"].astype(float) # MA200 Daily df["MA200"] = ta.trend.sma_indicator( df["close"], window=200 ) last = df.iloc[-1] ma200 = last["MA200"] if pd.isna(ma200): continue high = last["high"] low = last["low"] close = last["close"] # Menyentuh MA200 touched = low <= ma200 <= high if touched: distance_pct = abs(close - ma200) / ma200 * 100 results.append({ "Coin": symbol, "Close": round(close, 6), "MA200": round(ma200, 6), "Distance %": round(distance_pct, 3) }) except Exception: pass # Urutkan terdekat results = sorted(results, key=lambda x: x["Distance %"]) print("\n===================================") print("COIN BARU MENYENTUH MA200 DAILY") print("===================================\n") if len(results) == 0: print("Tidak ada coin yang menyentuh MA200 hari ini.") else: for i, r in enumerate(results[:30], start=1): print( f"{i:02d}. {r['Coin']:12} " f"Close={r['Close']} " f"MA200={r['MA200']} " f"Dist={r['Distance %']}%" )
