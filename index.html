import requests import pandas as pd from ta.trend import SMAIndicator BINANCE_URL = "https://api.binance.com" def get_symbols(): data = requests.get( f"{BINANCE_URL}/api/v3/exchangeInfo", timeout=20 ).json() symbols = [] for s in data["symbols"]: if ( s["status"] == "TRADING" and s["quoteAsset"] == "USDT" and "SPOT" in s["permissions"] ): symbols.append(s["symbol"]) return symbols def get_klines(symbol): try: data = requests.get( f"{BINANCE_URL}/api/v3/klines", params={ "symbol": symbol, "interval": "1d", "limit": 250 }, timeout=20 ).json() if not isinstance(data, list): return None df = pd.DataFrame(data) df = df.iloc[:, :6] df.columns = [ "time", "open", "high", "low", "close", "volume" ] for col in ["open", "high", "low", "close", "volume"]: df[col] = df[col].astype(float) return df except: return None def fresh_touch_ma200(df): if len(df) < 210: return None df["ma200"] = SMAIndicator( close=df["close"], window=200 ).sma_indicator() today = df.iloc[-1] yesterday = df.iloc[-2] ma_today = today["ma200"] if pd.isna(ma_today): return None touch_today = ( today["low"] <= ma_today <= today["high"] ) if not touch_today: return None ma_yesterday = yesterday["ma200"] touch_yesterday = ( yesterday["low"] <= ma_yesterday <= yesterday["high"] ) if touch_yesterday: return None for i in range(-12, -2): row = df.iloc[i] ma = row["ma200"] if pd.isna(ma): continue if row["low"] <= ma <= row["high"]: return None distance = abs( today["close"] - ma_today ) / ma_today * 100 return { "close": today["close"], "ma200": ma_today, "distance": distance, "volume": today["volume"] } def main(): symbols = get_symbols() results = [] print(f"Scanning {len(symbols)} coins...") for symbol in symbols: df = get_klines(symbol) if df is None: continue result = fresh_touch_ma200(df) if result: results.append({ "symbol": symbol, **result }) print(f"FOUND -> {symbol}") results = sorted( results, key=lambda x: x["distance"] ) print("\n") print("=" * 100) print("FRESH TOUCH MA200 DAILY") print("=" * 100) if not results: print("No coins found.") return for i, r in enumerate(results[:30], start=1): print( f"{i:02d}. " f"{r['symbol']:12} " f"Close={r['close']:.6f} " f"MA200={r['ma200']:.6f} " f"Distance={r['distance']:.2f}% " f"Volume={r['volume']:.0f}" ) if __name__ == "__main__": main()
